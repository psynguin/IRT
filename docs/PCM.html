import {slider} from "@jashkenas/inputs";
viewof t1 = slider({min: -4, max: 4, step: 0.1, value: -1.5, title: "Threshold 1 (tau 1)"});
viewof t2 = slider({min: -4, max: 4, step: 0.1, value: 0, title: "Threshold 2 (tau 2)"});
viewof t3 = slider({min: -4, max: 4, step: 0.1, value: 1.5, title: "Threshold 3 (tau 3)"});
md`
  ${tex.block`
    \begin{aligned}
    \text{total} &= 1 + e^{(\theta - ${t1})} + e^{(2\theta - ${t1} - ${t2})} + e^{(3\theta - ${t1} - ${t2} - ${t3})}
    \end{aligned}
  `}
`
md`
  ${tex.block`
    \begin{aligned}
    P(Y &= 0|\theta) &= \frac{1}{total} \\
    P(Y &= 1|\theta) &= \frac{e^{(\theta - ${t1})}}{total} \\
    \end{aligned}
  `}
`
md`
  ${tex.block`
    \begin{aligned}
    P(Y &= 2|\theta) &= \frac{e^{(2\theta - ${t1} - ${t2})}}{total} \\
    P(Y &= 3|\theta) &= \frac{e^{(3\theta - ${t1} - ${t2} - ${t3})}}{total}
    \end{aligned}
  `}
`
x = Array.from({ length: (6 - (-6)) / 0.01 + 1 }, (_, i) => -6 + i * 0.01);

// Thresholds (delta parameters)
deltas = [0, t1, t2, t3]; // d0 is always 0 for the reference category

// Calculate PCM Probabilities
results = x.map(theta => {
  // 1. Calculate the exponent for each category k: sum_{j=0}^k (theta - delta_j)
  let exponents = deltas.map((_, k) => {
    let sum = 0;
    for (let j = 0; j <= k; j++) {
      sum += (theta - deltas[j]);
    }
    return Math.exp(sum);
  });

  // 2. Calculate the denominator (sum of all exponents)
  let denominator = exponents.reduce((a, b) => a + b, 0);

  // 3. Normalize to get probabilities
  return exponents.map(exp => exp / denominator);
});

// Extract for plotting
y0 = results.map(r => r[0]);
y1 = results.map(r => r[1]);
y2 = results.map(r => r[2]);
y3 = results.map(r => r[3]);

// Plot
Plot.plot({
  marks: [
    Plot.ruleY([0]),
    Plot.ruleX([0]),
    // Categories
    Plot.line(x.map((t, i) => ({ x: t, y: y0[i] })), { x: "x", y: "y", stroke: "#e41a1c" }),
    Plot.line(x.map((t, i) => ({ x: t, y: y1[i] })), { x: "x", y: "y", stroke: "#377eb8" }),
    Plot.line(x.map((t, i) => ({ x: t, y: y2[i] })), { x: "x", y: "y", stroke: "#4daf4a" }),
    Plot.line(x.map((t, i) => ({ x: t, y: y3[i] })), { x: "x", y: "y", stroke: "#984ea3" }),
    
    // Threshold markers (In PCM, thresholds are intersections of adjacent curves)
    Plot.dot([
      {x: t1, y: y0[x.findIndex(t => t >= t1)]}, // Intersection of y0 and y1
      {x: t2, y: y1[x.findIndex(t => t >= t2)]}, // Intersection of y1 and y2
      {x: t3, y: y2[x.findIndex(t => t >= t3)]}  // Intersection of y2 and y3
    ], { x: "x", y: "y", fill: "black", r: 4 })
  ],
  x: { label: "Capacidad (θ)", domain: [-6, 6], grid: true },
  y: { label: "Probabilidad", domain: [0, 1], grid: true }
})
